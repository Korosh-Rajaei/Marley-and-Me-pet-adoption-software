    def SearchTextFile(self,text,path):
        if os.path.isfile(path):
            f = open(path, "r")
            data = f.readlines()
            for i in data:
                if text in i:
                    # list2=i.split(" ||||| ")
                    s=i.replace("\n","")
                    f.close()
                    print("found the text/description!")
                    return s
            print("the text/description doesn't exist!")
            return False
        else:
            print("Text file does not exist!:\n")
            return False
